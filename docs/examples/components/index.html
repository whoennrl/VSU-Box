    Компоненты и интерактивность

    Экран с готовыми UI-компонентами: поле ввода, кнопка сохранения, отображение сохранённого текста. Демонстрирует использование sdk.ui.components и sdk.ui.onAction.

    Разрешения: storage

    // index.js
VSUBoxSDK.ready(async (sdk) => {
    const c   = sdk.ui.components
    const sid = await sdk.ui.createScreen({ title: "Быстрые заметки" })
    await sdk.ui.addMenuButton({ label: "Заметки", screenId: sid })

    async function render() {
        const text = (await sdk.storage.get("note")) ?? ""
        await sdk.ui.setContent(sid,
            `<div style="padding:20px;font-family:Nunito,sans-serif">` +
            c.heading("Моя заметка") +
            c.input({ id: "note-text", placeholder: "Напишите что-нибудь...", value: text }) +
            c.spacer(10) +
            c.button("Сохранить", { id: "save-note", primary: true }) +
            c.divider() +
            c.text(text || "(пусто)", { muted: !text }) +
            `</div>`
        )
    }

    sdk.ui.onAction("save-note", async (value) => {
        if (!value?.trim()) { sdk.ui.toast("Введите текст"); return }
        await sdk.storage.set("note", value.trim())
        sdk.ui.toast("Заметка сохранена!")
        await render()
    })

    await render()
    await sdk.ui.navigate(sid)
})
